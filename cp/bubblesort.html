This is an old HackerRank problem that I found through usaco.guide.

The problem asks us to perform a bubble sort on a series of numbers.

This problem is incredibly trivial. Here's the code for this problem:

void countSwaps(vector<int> a) {
    int ans = 0;
    for (int i = 0; i < (int) a.size(); i++) {

        for (int j = 0; j < (int) a.size() - 1; j++) {
            // Swap adjacent elements if they are in decreasing order
            if (a[j] > a[j + 1]) {
                swap(a[j], a[j + 1]);
                ans += 1;
            }
        }

    }
    cout << "Array is sorted in " << ans << " swaps.\n";
    cout << "First Element: " << a[0] << "\n";
    cout << "Last Element: " << a[(int) a.size() - 1] << "\n";
    return;
}
