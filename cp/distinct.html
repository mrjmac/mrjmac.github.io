This is the 20th problem in the CSES competitive programming problem set.

The problem goes like this: given a list of numbers, identify how many unique numbers there are.

There are two basic ways to do this problem. First of all, you could add all the numbers to a set and print out it's size. Alternatively, you could instead sort the numbers and add one to your answer each time adjacent numbers are different values. Here's the code for both of these solutions:

sort: 

int main()
{
    cin.tie(0)->sync_with_stdio(0);

    int n, ans = 1;
    cin >> n;

    vector<int> nums(n);

    for (int i = 0; i < n; i++)
    {
        cin >> nums[i];
    }

    sort(nums.begin(), nums.end());

    for (int i = 1; i < n; i++)
    {
        if (nums[i] != nums[i - 1])
        {
            ans += 1;
        }
    }

    cout << ans << "\n";
}

set:

int main()
{
    cin.tie(0)->sync_with_stdio(0);

    int n;
    set<int> ans;
    cin >> n;

    for (int i = 0; i < n; i++)
    {
        int a;
        cin >> a;
        ans.insert(a);
    }

    cout << ans.size() << "\n";
}
