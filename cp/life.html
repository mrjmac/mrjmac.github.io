Lifeguards

This is an old usaco bronze problem that I found through usaco.guide.

This problem is rather long like all usaco problems, so here's a link so you can read it.

This problem is pretty basic, because N is 100 at max and each time period is 1000 units, we can fire each lifeguard and simulate how much time is covered without them. Here's the code for this problem:

int n, a, b, maxCover, numCover, covered[1000];
vector<pair<int, int>> shift;

int main()
{
    cin.tie(0)->sync_with_stdio(0);
    freopen("lifeguards.in", "r", stdin);
    freopen("lifeguards.out", "w", stdout);

    cin >> n;
    shift.resize(n);

    for (int i = 0; i < n; i++)
    {
        cin >> a >> b;
        shift[i] = make_pair(a, b);
        for (int j = a; j < b; j++)
        {
            covered[j]++;
        }
    }

    for (pair<int, int> worker : shift)
    {
        for (int i = worker.first; i < worker.second; i++)
        {
            covered[i]--;
        }

        numCover = 0;

        for (int i = 0; i < 1000; i++)
        {
            if (covered[i] > 0)
            {
                numCover++;
            }
        }

        for (int i = worker.first; i < worker.second; i++)
        {
            covered[i]++;
        }

        maxCover = max(maxCover, numCover);
    }

    cout << maxCover << "\n";
}
