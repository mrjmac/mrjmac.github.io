This problem isn't too bad once your break it down. In order to tell if a solution is unique, we can just insert it into a set. This means that we can use recursive backtracking to emulate every possible setup that could've happened. Here's the code for this problem:

void find_permutations(int day, int a, vector<int> newnums, int b, vector<int> newnums2)
{
    if (day == 4)
    {
        ans.insert(a);
        return;
    }
    for (int i = 0; i < (int) newnums.size(); i++)
    {
        int curr = newnums[i];

        vector<int> tempA = newnums;
        tempA.erase(begin(tempA) + i);
        vector<int> tempB = newnums2;
        tempB.push_back(curr);

        find_permutations(day + 1, b + curr, tempB, a - curr, tempA);
    }
}
