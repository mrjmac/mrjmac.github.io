This problem was pretty simple. In order to easily brute force this in 1 pass, all we have to do is sort the array. Then we can check every combination and easily meet the time constraints.

vector<int> sizes(n);
for (int i = 0; i < n; i++)
{
    cin >> sizes[i];
}

sort(sizes.begin(), sizes.end());

for (int i = 0; i < n; i++)
{
    int count = 1;
    for (int j = i + 1; j < n; j++)
    {
        if (sizes[j] - sizes[i] <= k)
        {
            count += 1;
        }
        else
        {
            break;
        }
    }
    ans = max(count, ans);
}

cout << ans << "\n";
