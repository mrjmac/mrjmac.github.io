This problem is pretty easy once you draw out all possible solutions for a test case. If we have no single kayaks, we just have to group people by how close they are in weight. This can be accomplished by sorting our list. However, we have 2 single kayaks. This is not that big of a deal, as we can just try out every possible combination of people in a single kayak to find the minimum weight. This will still run in time. Here's the code for this problem: 

int main()
{
    cin.tie(0)->sync_with_stdio(0);

    int n;
    cin >> n;

    n *= 2;

    vector<int> people(n);
    for (int i = 0; i < n; i++)
    {
        cin >> people[i];
    }

    sort(people.begin(), people.end());

    int ans = INT32_MAX;
    for (int i = 0; i < n; i++)
    {
        for (int j = i + 1; j < n; j++)
        {
            vector<int> people2;
            for (int l = 0; l < n; l++)
            {
                if (l != i && l != j)
                {
                    people2.push_back(people[l]);
                }
            }

            int curr = 0;
            for (int l = 0; l < n - 2; l += 2)
            {
                curr += people2[l + 1] - people2[l];
            }
            ans = min(curr, ans);
        }
    }
    cout << ans << "\n";
}
